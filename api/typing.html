Typing

sisl will use typing to enable static analyzers and to aid users when reading documentation.

The typing types is shown below:

AtomsArgument

alias of Optional[Union[ndarray[Any, dtype[Union[int32, int64, bool_]]], int, str, dict, Atom, AtomCategory, GenericCategory, Shape, bool]]

AtomsLike

alias of Union[int, str, Atom, Sequence[Union[int, str, Atom]], Atoms]

Axis

alias of int

Axies

alias of Union[int, Sequence[int]]

Axes

alias of Sequence[int]

Coord

alias of Sequence[float]

CoordOrScalar

alias of Union[float, Sequence[float]]

FuncType

alias of Callable[[...], Any]

GeometryLike

alias of Union[str, BaseSile, Path, Geometry]

GridLike

alias of Union[str, BaseSile, Path, Grid]

IndexArgument

alias of Union[ndarray[Any, dtype[Union[int32, int64, bool_]]], int]

KPoint

alias of Sequence[float]

LatticeLike

alias of Union[str, BaseSile, Path, Lattice]

LatticeOrGeometry

alias of Union[Lattice, Geometry]

LatticeOrGeometryLike

alias of Union[str, BaseSile, Path, Lattice, Geometry]

OrbitalsArgument

alias of Optional[Union[ndarray[Any, dtype[Union[int32, int64, bool_]]], int, str, dict, AtomCategory, Shape, bool]]

SileLike

alias of Union[str, BaseSile, Path]

SparseMatrix

alias of Union[spmatrix, sparray, SparseCSR]

SparseMatrixExt

alias of Union[spmatrix, sparray]
